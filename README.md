### Installation
`$npm install -S amanhimself`

### Usage
```javascript
const aman = require('amanhimself');

console.log(aman.name.first);
//=> 'aman'
console.log(aman.projects.jade.description);
//=> Jade/Pug snippets package for Atom.
```
